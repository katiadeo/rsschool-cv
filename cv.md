// Check if a string or a number is a palindrome 
function isPalindrome(line) {
	line = line.toString();
	for (let i = 0; i < line.length; i++) {
		if (line[i] !== line[line.length - i - 1]) {
  		return false;
  	}
    return true;
  }
}

console.log(isPalindrome(123321));
